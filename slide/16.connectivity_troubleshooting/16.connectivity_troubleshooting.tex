\begin{frame}
	\frametitle{Extended Ping}
	\begin{itemize}
		\item Scenario: The user on PC1 complains that he can’t access services on PC3
		\item The problem is R4 does not have a route to 10.0.1.0/24
		\item Traffic which originates from a router always uses the IP address on theoutgoing interface as the source address
		\item A ping from R1 to 10.1.2.10 will succeed because R4 has a route to10.0.0.1
	\end{itemize}
\end{frame}
